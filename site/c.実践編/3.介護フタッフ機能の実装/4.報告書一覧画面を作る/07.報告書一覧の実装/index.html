    import { ReportBase } from '@/contexts/work-reports/entities/reports/reportSchema';
import React, { useMemo } from 'react';
import { ReportRow } from '../ReportRow';

type Props = {
  reports: ReportBase[];
};

export const ReportList: React.FC<Props> = (props) => {
  const { reports } = props;

  const reportElements = useMemo(() => {
    return reports.map((report) => {
      return <ReportRow key={report.id} report={report} />;
    });
  }, [reports]);

  return <div>{reportElements}</div>;
};
